Step 1: Create Folder Structure
In VS Code Terminal (Ctrl + `):

bash
mkdir src tests rules samples docs
mkdir src/CodeMigrationTool.Core
mkdir src/CodeMigrationTool.Analyzers
mkdir src/CodeMigrationTool.Transformers
mkdir src/CodeMigrationTool.Models
mkdir src/CodeMigrationTool.Database
mkdir src/CodeMigrationTool.Services
mkdir src/CodeMigrationTool.CLI
mkdir tests/CodeMigrationTool.Analyzers.Tests
mkdir tests/CodeMigrationTool.Transformers.Tests
mkdir tests/CodeMigrationTool.Integration.Tests
Step 2: Create Solution & Projects
bash
dotnet new sln -n CodeMigrationTool
dotnet new classlib -n CodeMigrationTool.Core -o src/CodeMigrationTool.Core
dotnet new classlib -n CodeMigrationTool.Analyzers -o src/CodeMigrationTool.Analyzers
dotnet new classlib -n CodeMigrationTool.Transformers -o src/CodeMigrationTool.Transformers
dotnet new classlib -n CodeMigrationTool.Models -o src/CodeMigrationTool.Models
dotnet new classlib -n CodeMigrationTool.Database -o src/CodeMigrationTool.Database
dotnet new classlib -n CodeMigrationTool.Services -o src/CodeMigrationTool.Services
dotnet new console -n CodeMigrationTool.CLI -o src/CodeMigrationTool.CLI

dotnet new xunit -n CodeMigrationTool.Analyzers.Tests -o tests/CodeMigrationTool.Analyzers.Tests
dotnet new xunit -n CodeMigrationTool.Transformers.Tests -o tests/CodeMigrationTool.Transformers.Tests
dotnet new xunit -n CodeMigrationTool.Integration.Tests -o tests/CodeMigrationTool.Integration.Tests

# Add all to solution
dotnet sln add src/*/CodeMigrationTool.*.csproj
dotnet sln add tests/*/CodeMigrationTool.*.Tests.csproj
Step 3: Use Copilot to Add NuGet Packages
In VS Code, open the Terminal and run:

bash
# Database project
cd src/CodeMigrationTool.Database
dotnet add package Microsoft.EntityFrameworkCore.SqlServer --version 8.0.0
dotnet add package Microsoft.EntityFrameworkCore.Tools --version 8.0.0
cd ../..

# Analyzers project
cd src/CodeMigrationTool.Analyzers
dotnet add package Microsoft.CodeAnalysis.CSharp --version 4.6.0
dotnet add package Microsoft.Build --version 17.6.0
cd ../..

# CLI project
cd src/CodeMigrationTool.CLI
dotnet add package Spectre.Console --version 0.48.0
cd ../..

# Test projects
cd tests/CodeMigrationTool.Analyzers.Tests
dotnet add package Moq --version 4.18.0
dotnet add package Shouldly --version 4.1.0
cd ../../..
